# Object Types / data Types

- Number : 1234, 3.145, 3+4j, 0b11, fraction()

- String : 'spam', "BOB"

- List :  [ 1,[2,'three'], 4.5], list(range(10))

- Tuple : (1, 'spam', 4, 'U'), tuple('spam'), namedtuple

- Dictionlary : {'food': 'spam', 'tast': 'yum'}

- Set : set('abc'), {'a','b','c'}

- File : open ('eggs.txt'), open(r'C:\ham.bin', 'wb')

- Boolean : true, False

- None : None
